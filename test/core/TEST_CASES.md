### 测试框架建议

```python
# 示例: 使用 pytest
import pytest
from core.utils import ElevationFilter

def test_gpx_filter():
    """测试 GPX 滤波"""
    # 准备测试数据
    elevations = [100, 102, 98, 105, 103]

    # 执行滤波
    smoothed = ElevationFilter.smooth(
        elevations,
        FilterConfig.GPX
    )

    # 验证结果
    assert len(smoothed) == len(elevations)
    assert max(smoothed) - min(smoothed) < 10
```
